/* =========================
   ROM LIBRARY (EDIT THIS)
========================= */
const ROMS = [
  {
    name: "Super Mario 64 (Hack)",
    url: "https://example.com/sm64.z64"
  },
  {
    name: "B3313",
    url: "https://example.com/b3313.z64"
  }
];

let romName = "default";

/* =========================
   LIBRARY UI
========================= */
function toggleLibrary() {
  const lib = document.getElementById("library");
  lib.style.display = lib.style.display === "block" ? "none" : "block";
  renderLibrary();
}

function renderLibrary() {
  const lib = document.getElementById("library");
  lib.innerHTML = "<h2 style='color:white'>Retro64 Library</h2>";

  ROMS.forEach(r => {
    const b = document.createElement("button");
    b.className = "gameBtn";
    b.innerText = r.name;
    b.onclick = () => loadROM(r.url, r.name);
    lib.appendChild(b);
  });
}

/* =========================
   EMULATOR START (FIXED)
========================= */
function loadROM(url, name) {
  romName = name;

  document.getElementById("library").style.display = "none";
  document.getElementById("emulator").innerHTML = "";

  window.EJS_player = "#emulator";
  window.EJS_core = "n64";
  window.EJS_gameUrl = url;
  window.EJS_pathtodata = "https://cdn.emulatorjs.org/latest/data/";

  setTimeout(() => {
    const s = document.createElement("script");
    s.src = "https://cdn.emulatorjs.org/latest/data/loader.js";
    document.body.appendChild(s);
  }, 300);
}

/* =========================
   INPUT SYSTEM
========================= */
function press(btn){
  const map = {
    A:"KeyX", B:"KeyZ", Z:"KeyC", Start:"Enter",
    CUp:"Numpad8", CDown:"Numpad2",
    CLeft:"Numpad4", CRight:"Numpad6"
  };

  const code = map[btn];
  if(!code) return;

  document.dispatchEvent(new KeyboardEvent("keydown",{code}));
  setTimeout(()=>document.dispatchEvent(new KeyboardEvent("keyup",{code})),80);
}

/* =========================
   ANALOG STICK
========================= */
let drag=false;
const stick=document.getElementById("stick");
const knob=document.getElementById("knob");

stick.addEventListener("touchstart",e=>{drag=true;e.preventDefault();},{passive:false});
stick.addEventListener("touchend",()=>{drag=false;knob.style.transform="translate(0,0)";});
stick.addEventListener("touchmove",e=>{
  if(!drag)return;
  e.preventDefault();

  const t=e.touches[0];
  const r=stick.getBoundingClientRect();

  let x=t.clientX-r.left-55;
  let y=t.clientY-r.top-55;

  x=Math.max(-40,Math.min(40,x));
  y=Math.max(-40,Math.min(40,y));

  knob.style.transform=`translate(${x}px,${y}px)`;

  if(y<-20)press("Up");
  if(y>20)press("Down");
  if(x<-20)press("Left");
  if(x>20)press("Right");
},{passive:false});
